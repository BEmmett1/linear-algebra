\begin{problem}{E1}
Write an augmented matrix corresponding to the following system of linear equations.
\begin{align*}
x_1+3x_2-4x_3 +x_4 &= 5 \\
3x_1+9x_2+x_3-7x_4 &= 0 \\
x_1-x_3 +x_4 &= 1
\end{align*}
\end{problem}
\begin{solution}
\[
\begin{bmatrix}[cccc|c]
1 & 3 & -4 & 1 & 5 \\
3 & 9 & 1 & -7 & 0 \\
1 & 0 & -1 & 1 &  1
\end{bmatrix}
\]
\end{solution}
