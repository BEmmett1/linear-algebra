    1  \\
    2\\
    1
  \end{bmatrix}\),
  \(S(\vec e_3)=\begin{bmatrix}

    0
  \end{bmatrix}\), and
  \(S(\vec e_4)=\begin{bmatrix}
    3 \\
    2\\
    1
  \end{bmatrix}\),
\item

  \(T(\vec e_2)=\begin{bmatrix}
     1 \\
     0 \\
     4 \\
  \end{bmatrix}\), and
  \(T(\vec e_3)=\begin{bmatrix}
     1 \\
     2 \\
     -3 \\
  \end{bmatrix}\).
\end{enumerate}
\end{problem}
