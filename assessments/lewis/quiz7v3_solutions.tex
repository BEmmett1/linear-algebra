\begin{problem}{E2}
Find \(\RREF A\), where
\[
  A =
  \begin{bmatrix}[cccc|c]
    3 & -2 & 1 & 8 & -5 \\
    2 & 2 & 0 & 6 & -2 \\
    -1 & 1 & 1 & -4 & 6 \\
  \end{bmatrix}
\]
\end{problem}
\begin{solution}
\[
  \RREF A =
  \begin{bmatrix}[cccc|c]
    1 & 0 & 0 & 3 & -2 \\
    0 & 1 & 0 & 0 & 1 \\
    0 & 0 & 1 & -1 & 3
  \end{bmatrix}
\]
\end{solution}
