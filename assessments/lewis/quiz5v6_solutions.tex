\begin{problem}{E2}
Find \(\RREF A\), where
\[
  A =
  \begin{bmatrix}[ccc|c]
    2 & -1 & 5 & 4 \\
    -1 & 0 & -2 & -1 \\
    1 & 3 & -1 & -5
  \end{bmatrix}
\]
\end{problem}
\begin{solution}
\[
  \RREF A =
  \begin{bmatrix}[ccc|c]
    1 & 0 & 2 & 1 \\
    0 & 1 & -1 & -2 \\
    0 & 0 & 0 & 0
  \end{bmatrix}
\]
\end{solution}
