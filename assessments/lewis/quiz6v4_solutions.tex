\begin{problem}{E1}
Write a system of linear equations corresponding to the following
augmented matrix.
\[
\begin{bmatrix}[cccc|c]
3 & -1 & 0 & 1 & 5 \\
-1 & 9 & 1 & -7 & 0 \\
1 & 0 & -1 & 0 & -3
\end{bmatrix}
\]
\end{problem}
\begin{solution}
\begin{align*}
3x_1-x_2+x_4 &= 5 \\
-x_1+9x_2+x_3-7x_4 &= 0 \\
x_1-x_3 &= -3
\end{align*}
\end{solution}

\begin{problem}{E3}
Solve the system of linear equations.
\begin{align*}
2x+y-z+w &=5 \\
3x-y-2w &= 0 \\
-x+5z+3w&=-1
\end{align*}
\end{problem}
\begin{solution}
$$\RREF\left( \begin{bmatrix}[cccc|c] 2 & 1 & -1 & 0 & 5 \\ 3 & -1 & 0 & -2 & 0 \\ -1 & 0 & 5 & 0 & -1 \end{bmatrix} \right) = \begin{bmatrix}[cccc|c] 1 & 0 & 0 & -\frac{1}{12} & 1 \\ 0 & 1 & 0 & \frac{7}{4} & 3 \\ 0 & 0 & 1 & \frac{7}{12} & 0 \end{bmatrix}$$
So the solutions are $$\left\{ \begin{bmatrix} 1+a \\ 3-21a \\ -7a \\ 12a \end{bmatrix}\ \big|\ a \in \IR\right\}$$
\end{solution}

\begin{problem}{E4}
Find a basis for the solution set to the homogeneous system of equations
\begin{align*}
2x_1+3x_2-5x_3+14x_4 &= 0\\
x_1+x_2-x_3+5x_4 &= 0
\end{align*}
\end{problem}
\begin{solution}
Let \(A =
  \begin{bmatrix}[cccc|c]
    2 & 3 & -5 & 14 & 0 \\
    1 & 1 & -1 & 5 & 0
  \end{bmatrix}
\), so \(\RREF A =
  \begin{bmatrix}[cccc|c]
    1 & 0 & 2 & 1 & 1 \\
    0 & 1 & -3 & 4 & 2 \\
  \end{bmatrix}
\).
It follows that the basis for the solution set is given by \(\left\{
  \begin{bmatrix}
    - 2 \\
    3 \\
    1 \\
    0
  \end{bmatrix},
  \begin{bmatrix}
    -1 \\
    - 4 \\
    0 \\
    1
  \end{bmatrix}
\right\}\).
\end{solution}
