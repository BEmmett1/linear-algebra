\begin{problem}{E1}
Write a system of linear equations corresponding to the following
augmented matrix.
\[
\begin{bmatrix}[ccc|c]
-4 & -1 & 3 & 2  \\
1 & 2 & -1 & 0 \\
-1 & 4 & 1 & 4
\end{bmatrix}
\]
\end{problem}
\begin{solution}
\begin{align*}
-4x_1-x_2+3x_3&=2 \\
x_1+2x_2-x_3 &= 0 \\
-x_1+4x_2+x_3&=4
\end{align*}
\end{solution}
