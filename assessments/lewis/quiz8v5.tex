\begin{problem}{E1}
Write a system of linear equations corresponding to the following
augmented matrix.
\[
\begin{bmatrix}[cccc|c]
3 & -1 & 0 & 1 & 5 \\
-1 & 9 & 1 & -7 & 0 \\
1 & 0 & -1 & 0 & -3
\end{bmatrix}
\]
\end{problem}

\begin{problem}{E3}

\begin{align*}
2x_1-2x_2+6x_3-x_4 &=-1 \\
3x_1+6x_3+x_4 &= 5 \\
-4x_1+x_2-9x_3+2x_4&=-7
\end{align*}
\end{problem}

\begin{problem}{E4}

\begin{align*}
4x_1+4x_2+3x_3-6x_4 &= 0 \\
-2x_3-4x_4 &= 0 \\

\begin{problem}{V1}
Let $V$ be the set of all points on the line $x+y=2$ with the operations, for any $(x_1,y_1), (x_2,y_2) \in V$, $c \in \IR$,
\begin{align*}
(x_1,y_1) \oplus (x_2,y_2) &= (x_1+x_2-1,y_1+y_2-1) \\
c \odot (x_1,y_1) &= (cx_1-(c-1), cy_1-(c-2))
\end{align*}
Determine if $V$ is a vector space or not.
\end{problem}
