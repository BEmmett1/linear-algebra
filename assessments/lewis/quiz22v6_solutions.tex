\begin{problem}{A1}
Let $T: \IR^3 \rightarrow \IR$ be the linear transformation given by $$T\left(\begin{bmatrix} x_1 \\ x_2 \\ x_3  \end{bmatrix} \right) = \begin{bmatrix} x_2+3x_3 \end{bmatrix}.$$ Write the matrix for $T$ with respect to the standard bases of $\IR^3$ and $\IR$.
\end{problem}
\begin{solution}
$$\begin{bmatrix} 0 & 1 & 3 \end{bmatrix}$$
\end{solution}

\end{problem}

\begin{problem}{M1}
Let 
\begin{align*}
C &= \begin{bmatrix} 2 & 3 \\ 0 & 1 \end{bmatrix} & D&= \begin{bmatrix} 3 & 1 & 0 \end{bmatrix} & E&= \begin{bmatrix} 2 & 0 \\ 0 & -1 \\ 1 & -1 \end{bmatrix}
\end{align*}

Determine which of the six products $CD$, $CE$, $DC$, $DE$, $EC$, $ED$ can be computed, and compute them.
\end{problem}
\begin{solution}
$$EC=\begin{bmatrix} 4 & 6 \\ 0 & -1 \\ 2 & 2 \end{bmatrix}$$
$$DE = \begin{bmatrix} 6 & -1 \end{bmatrix}$$
\end{solution}
