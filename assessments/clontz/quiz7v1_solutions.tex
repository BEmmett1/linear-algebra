\begin{problem}{E2}
Put the following matrix in reduced row echelon form.
$$\begin{bmatrix}[ccc|c] -3 & 1 & 0 & 2 \\ -8 & 2 & -1 & 6 \\ 0 & 2 & 3 & -2 \end{bmatrix}$$
\end{problem}
\begin{solution}
$$\begin{bmatrix}[ccc|c] 
-3 & 1 & 0 & 2 \\
 -8 & 2 & -1 & 6 \\
 0 & 2 & 3 & -2 
\end{bmatrix} \sim
\begin{bmatrix}[ccc|c] 
1 & -\frac{1}{3} & 0 & -\frac{2}{3} \\
 -8 & 2 & -1 & 6 \\
 0 & 2 & 3 & -2 
\end{bmatrix} \sim
\begin{bmatrix}[ccc|c] 
1 & -\frac{1}{3} & 0 & -\frac{2}{3} \\
 0 & -\frac{2}{3} & -1 & \frac{2}{3} \\
 0 & 2 & 3 & -2 
\end{bmatrix} \sim
\begin{bmatrix}[ccc|c] 
1 & -\frac{1}{3} & 0 & -\frac{2}{3} \\
 0 & 1 & \frac{3}{2} & -1 \\
 0 & 2 & 3 & -2 
\end{bmatrix} \sim
\begin{bmatrix}[ccc|c] 
1 & 0 & \frac{1}{2} & -1 \\
 0 & 1 & \frac{3}{2} & -1 \\
 0 & 0 & 0 & 0 
\end{bmatrix}$$
\end{solution}
