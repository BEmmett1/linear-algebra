\begin{problem}{E1}
Write an augmented matrix corresponding to the following system of linear equations.
\begin{align*}
x_1+4x_3 &= 1 \\
x_2-x_3 &= 7 \\
x_1-x_2+3x_3 &= -1
\end{align*}
\end{problem}
\begin{solution}
$$\begin{bmatrix}[ccc|c]
1 & 0 & 4 & 1 \\
0 & 1 & -1 & 7 \\
1 & -1 & 3 & -1
\end{bmatrix}$$
\end{solution}

\begin{problem}{E3} 
Solve the system of linear equations.
\begin{align*}
2x+y-z+w &=5 \\
3x-y-2w &= 0 \\
-x+5z+3w&=-1
\end{align*}
\end{problem}
\begin{solution}
$$\RREF\left( \begin{bmatrix}[cccc|c] 2 & 1 & -1 & 0 & 5 \\ 3 & -1 & 0 & -2 & 0 \\ -1 & 0 & 5 & 0 & -1 \end{bmatrix} \right) = \begin{bmatrix}[cccc|c] 1 & 0 & 0 & -\frac{1}{12} & 1 \\ 0 & 1 & 0 & \frac{7}{4} & 3 \\ 0 & 0 & 1 & \frac{7}{12} & 0 \end{bmatrix}$$
So the solutions are $$\left\{ \begin{bmatrix} 1+a \\ 3-21a \\ -7a \\ 12a \end{bmatrix}\ \big|\ a \in \IR\right\}$$
\end{solution}
