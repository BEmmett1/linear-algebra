\begin{problem}{E2}
Put the following matrix in reduced row echelon form.
$$\begin{bmatrix}
 3 & -1 & 0 \\
 -1 & 0 & -1 \\
 -1 & 1 & 2 \\
 0 & 2 & 6
\end{bmatrix}$$
\end{problem}
\begin{solution}
$$\begin{bmatrix}
 3 & -1 & 0 \\
 -1 & 0 & -1 \\
 -1 & 1 & 2 \\
 0 & 2 & 6
\end{bmatrix}
\sim 
\begin{bmatrix}
 -1 & 0 & -1 \\
 3 & -1 & 0 \\
 -1 & 1 & 2 \\
 0 & 2 & 6
\end{bmatrix}
\sim 
\begin{bmatrix}
 1 & 0 & 1 \\
 3 & -1 & 0 \\
 -1 & 1 & 2 \\
 0 & 2 & 6
\end{bmatrix}
\sim
\begin{bmatrix}
 1 & 0 & 1 \\
 0 & -1 & -3 \\
 0 & 1 & 3 \\
 0 & 2 & 6
\end{bmatrix}
\sim
\begin{bmatrix}
 1 & 0 & 1 \\
 0 & 1 & 3 \\
 0 & -1 & -3 \\
 0 & 2 & 6
\end{bmatrix}
\sim
\begin{bmatrix}
 1 & 0 & 1 \\
 0 & 1 & 3 \\
 0 & 0 & 0 \\ 
0 & 0 & 0 
\end{bmatrix}$$
\end{solution}
