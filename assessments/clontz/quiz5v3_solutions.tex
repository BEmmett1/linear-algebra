\begin{problem}{E2}
Find \(\RREF A\), where
\[
  A =
  \begin{bmatrix}[cc|c]
    2 & -7 & 4 \\
    1 & -3 & 2 \\
    3 & 0 & 3
  \end{bmatrix}
\]
\end{problem}
\begin{solution}
\[
  \RREF A =
  \begin{bmatrix}[cc|c]
    1 & 0 & 0 \\
    0 & 1 & 0 \\
    0 & 0 & 1
  \end{bmatrix}
\]
\end{solution}
