\begin{problem}{E1}
Write an augmented matrix corresponding to the following system of linear equations.
\begin{align*}
x+3y-4z &= 5 \\
3x+9y+z &= 0 \\
x-z &= 1
\end{align*}
\end{problem}
\begin{solution}
$$\begin{bmatrix}[ccc|c]
1 & 3 & -4 & 5 \\
3 & 9 & 1 & 0 \\
1 & 0 & -1 & 1 
\end{bmatrix}$$
\end{solution}
