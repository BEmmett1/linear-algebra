\begin{problem}{V1}
Let $V$ be the set of all points on the line $x+y=2$ with the operations, for any $(x_1,y_1), (x_2,y_2) \in V$, $c \in \IR$,
\begin{align*}
(x_1,y_1) \oplus (x_2,y_2) &= (x_1+x_2-1,y_1+y_2-1) \\

\end{align*}
Determine if $V$ is a vector space or not.
\end{problem}
