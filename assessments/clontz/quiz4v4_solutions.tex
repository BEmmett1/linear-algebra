\begin{problem}{E1}
Write a system of linear equations corresponding to the following
augmented matrix.
\[
\begin{bmatrix}[ccc|c]
1 & 0 & 4 & 1 \\
0 & 1 & -1 & 7 \\
1 & -1 & 3 & -1
\end{bmatrix}
\]
\end{problem}
\begin{solution}
\begin{align*}
x_1+4x_3 &= 1 \\
x_2-x_3 &= 7 \\
x_1-x_2+3x_3 &= -1
\end{align*}
\end{solution}
