\begin{problem}{E1}
Write a system of linear equations corresponding to the following
augmented matrix.
\[
\begin{bmatrix}[cccc|c]
3 & -1 & 0 & 1 & 5 \\
-1 & 9 & 1 & -7 & 0 \\
1 & 0 & -1 & 0 & -3
\end{bmatrix}
\]
\end{problem}
\begin{solution}
\begin{align*}
3x_1-x_2+x_4 &= 5 \\
-x_1+9x_2+x_3-7x_4 &= 0 \\
x_1-x_3 &= -3
\end{align*}
\end{solution}
