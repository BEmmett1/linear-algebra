\begin{problem}{E1}
Write a system of linear equations corresponding to the following
augmented matrix.
\[
\begin{bmatrix}[ccc|c]
2 & -1 & 0 & 1  \\
-1 & 4 & 1 & -7  \\
1 & 2 & -1 & 0
\end{bmatrix}
\]
\end{problem}
\begin{solution}
\begin{align*}
2x_1-x_2&=1 \\
-x_1+4x_2+x_3&=-7 \\
x_1+2x_2-x_3 &= 0
\end{align*}
\end{solution}
