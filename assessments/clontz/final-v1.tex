\begin{problem}{V1}

\begin{align*}
(x_1,y_1) \oplus (x_2,y_2) &= (x_1+x_2-1,y_1+y_2-1) \\
c \odot (x_1,y_1) &= (cx_1-(c-1), cy_1-(c-1))

\end{enumerate}
