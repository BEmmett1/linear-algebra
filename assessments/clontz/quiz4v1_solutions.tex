\begin{problem}{E1}
Write an augmented matrix corresponding to the following system of linear equations.
\begin{align*}
x_1+4x_3 &= 1 \\
x_2-x_3 &= 7 \\
x_1-x_2+3x_4 &= -1
\end{align*}
\end{problem}
\begin{solution}
$$\begin{bmatrix}[cccc|c]
1 & 0 & 4 & 0 & 1 \\
0 & 1 & -1 & 0 & 7 \\
1 & -1 & 0 & 3 & -1
\end{bmatrix}$$
\end{solution}
