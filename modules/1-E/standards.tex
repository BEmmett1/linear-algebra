\moduleQuestion{How can we solve systems of linear equations?}
\begin{standardsList}
  \standard{Systems as matrices}{SysMat}{
    translate back and forth between a system of linear equations
    and the corresponding augmented matrix.
  }
  \standard{Row reduction}{Rref}{
    put a matrix in reduced row echelon form.
  }
  \standard{Systems of linear equations}{SlvSys}{
    compute the solution set for a system of linear equations.
  }
\end{standardsList}
