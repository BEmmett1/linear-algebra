\begin{problem}{V10}
Find a basis for the solution space of the homogeneous system of equations
\begin{align*}
x+3y+3z+7w &= 0 \\
 x+3y-z-w &= 0 \\
  2x+6y+3z+8w &= 0 \\
   x+3y-2z-3w &= 0
\end{align*}
\end{problem}
\begin{solution}
A basis for the solution space is \[\left\{ \begin{bmatrix} 3 \\ -1 \\ 0 \\ 0 \end{bmatrix} , \begin{bmatrix} 1\\0 \\ 2 \\ -1 \end{bmatrix} \right\}\]
\end{solution}
