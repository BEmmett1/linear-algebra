
\begin{exerciseStatement}


Show how to compute the determinant of the matrix \[A= \left[\begin{array}{cccc}
3 & 2 & 3 & -5 \\
3 & -6 & 4 & 0 \\
-5 & -4 & -2 & 4 \\
0 & -3 & 0 & 1
\end{array}\right] .\].


\end{exerciseStatement}
    
\begin{exerciseAnswer} 
\[\operatorname{det}\  \left[\begin{array}{cccc}
3 & 2 & 3 & -5 \\
3 & -6 & 4 & 0 \\
-5 & -4 & -2 & 4 \\
0 & -3 & 0 & 1
\end{array}\right] = 104 \]
\end{exerciseAnswer}
