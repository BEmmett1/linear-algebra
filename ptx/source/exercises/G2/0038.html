.

\[\operatorname{det}\ \left[\begin{array}{cccc} -1 & -5 & -4 & 0 \\ -6 & -4 & -5 & 2 \\ 5 & -6 & 0 & 0 \\ -1 & 2 & -5 & 1 \end{array}\right] = -411 \]
