
\begin{exerciseStatement}


Show how to find the inverse of the matrix \(Q= \left[\begin{array}{cccc}
2 & -1 & 1 & 3 \\
0 & 1 & -1 & 8 \\
0 & 1 & 0 & 3 \\
1 & -1 & 1 & -2
\end{array}\right] \).


\end{exerciseStatement}
    
\begin{exerciseAnswer} 
\[Q^{-1}= \left[\begin{array}{cccc}
6 & -5 & 0 & -11 \\
3 & -3 & 1 & -6 \\
-5 & 4 & 1 & 10 \\
-1 & 1 & 0 & 2
\end{array}\right] \]
\end{exerciseAnswer}
