
\begin{exerciseStatement}


Show how to find the inverse of the matrix \(C= \left[\begin{array}{cccc}
1 & -1 & 4 & 2 \\
2 & -1 & 8 & -1 \\
0 & -2 & 1 & 8 \\
0 & 2 & -1 & -7
\end{array}\right] \).


\end{exerciseStatement}
    
\begin{exerciseAnswer} 
\[C^{-1}= \left[\begin{array}{cccc}
15 & -7 & -9 & -5 \\
-2 & 1 & 5 & 5 \\
-4 & 2 & 3 & 2 \\
0 & 0 & 1 & 1
\end{array}\right] \]
\end{exerciseAnswer}
