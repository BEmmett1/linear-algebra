
\begin{exerciseStatement}


Show how to find the inverse of the matrix \(B= \left[\begin{array}{cccc}
1 & 3 & 3 & -7 \\
0 & 1 & 2 & -4 \\
0 & 0 & 1 & -3 \\
0 & 3 & 5 & -8
\end{array}\right] \).


\end{exerciseStatement}
    
\begin{exerciseAnswer} 
\[B^{-1}= \left[\begin{array}{cccc}
1 & -15 & 7 & 4 \\
0 & 7 & -4 & -2 \\
0 & -9 & 4 & 3 \\
0 & -3 & 1 & 1
\end{array}\right] \]
\end{exerciseAnswer}
