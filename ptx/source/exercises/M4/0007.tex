
\begin{exerciseStatement}


Show how to find the inverse of the matrix \(M= \left[\begin{array}{cccc}
0 & -1 & -2 & -2 \\
-3 & 4 & 7 & -7 \\
-2 & 1 & -1 & -4 \\
1 & 0 & 1 & 4
\end{array}\right] \).


\end{exerciseStatement}
    
\begin{exerciseAnswer} 
\[M^{-1}= \left[\begin{array}{cccc}
-35 & -6 & -11 & -39 \\
-35 & -6 & -10 & -38 \\
11 & 2 & 3 & 12 \\
6 & 1 & 2 & 7
\end{array}\right] \]
\end{exerciseAnswer}
