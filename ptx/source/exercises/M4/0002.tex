
\begin{exerciseStatement}


Show how to find the inverse of the matrix \(P= \left[\begin{array}{cccc}
1 & -1 & 5 & 4 \\
1 & 0 & 4 & 0 \\
-1 & 1 & -4 & -3 \\
-1 & -1 & -6 & 2
\end{array}\right] \).


\end{exerciseStatement}
    
\begin{exerciseAnswer} 
\[P^{-1}= \left[\begin{array}{cccc}
4 & 9 & 8 & 4 \\
6 & 7 & 10 & 3 \\
-1 & -2 & -2 & -1 \\
2 & 2 & 3 & 1
\end{array}\right] \]
\end{exerciseAnswer}
