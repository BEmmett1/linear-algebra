
\begin{exerciseStatement}


Show how to find the inverse of the matrix \(N= \left[\begin{array}{cccc}
-2 & 1 & -2 & 0 \\
0 & 1 & 1 & 8 \\
1 & 0 & 1 & 2 \\
-1 & -1 & -1 & -5
\end{array}\right] \).


\end{exerciseStatement}
    
\begin{exerciseAnswer} 
\[N^{-1}= \left[\begin{array}{cccc}
3 & -1 & 9 & 2 \\
-3 & 0 & -10 & -4 \\
-5 & 1 & -14 & -4 \\
1 & 0 & 3 & 1
\end{array}\right] \]
\end{exerciseAnswer}
