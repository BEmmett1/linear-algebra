1 & -4 & -6 & 0 \\
0 & 1 & 0 & -2 \\
1 & -2 & -1 & 0 \\
0 & -5 & -4 & 7
\end{array}\right] \).


\end{exerciseStatement}

\end{exerciseAnswer}
