\moduleQuestion{What algebraic structure do matrices have?}
\begin{moduleStandards}
  \standard{Matrix Multiplication}{MatMlt}{
    multiply matrices.
  }
  \standard{Invertible Matrices}{InvVer}{
    determine if a square matrix is invertible or not.
  }
  \standard{Matrix inverses}{InvCmp}{
    compute the inverse matrix of an invertible matrix.
  }
  \standard{Row operations as multiplication}{RowOpMult}{
    describe the row reduction of a matrix as matrix multiplication.
  }
\end{moduleStandards}
