\begin{problem}{G3}
Find the eigenvalues of the matrix \(\begin{bmatrix} 5 & 1 \\ -24 & -6\end{bmatrix}\).
\end{problem}
\begin{solution}
Its characteristic polynomial is \(\lambda^2+\lambda-6=(\lambda+3)(\lambda-2)\), yielding the eigenvalues \(-3\) and \(2\).
\end{solution}
